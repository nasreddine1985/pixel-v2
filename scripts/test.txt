	/**
	 * @param fileName
	 * @param msgNbr
	 * @param position 
	 * @return parsing the msgNbr first Certis message from file "Filename"   
	 * @throws Exception 
	 * @throws TransformerException
	 */
	
	@SuppressWarnings("finally")
	public String[] parseCertisFile(String fileName , int msgNbr , int position ) 
	//	throws Exception 
		{
		// initialize  global variables
	
		
	    String[] ReturnValue = new String[2];
		try (BufferedReader in = new BufferedReader(new InputStreamReader(new FileInputStream(fileName),Charset.forName(getProperty(READER_ENCODING))))) {
			//in = new BufferedReader(new InputStreamReader(new FileInputStream(fileName),Charset.forName("ISO-8859-1")));


            Writer writer = new StringWriter();
		    out = new StreamResult(writer);
		    lineNumber=position;
		    
		    // initialize local variables
			String msgCertis = "";
			String str=null;
			int index = 0;
			boolean CertisMsgStart;
			
			// skip the firsts "position" lines 
			if( position ==0)   str=in.readLine();
			else for (int i = 0; i<=position; i++) str=in.readLine();
			
			// init the XML CERTIS result
			initXML();

			// read the firsts 'position' message Certis  
			while (index < msgNbr && !eof ){
				CertisMsgStart = true;	
				while (str != null  && CertisMsgStart ) {
					msgCertis +=  msgCertis.isEmpty()?str:DELIM+str ;
					str = in.readLine();
					if (str != null && str.startsWith(CERTIS_Msg_Starting)) CertisMsgStart=false;
					lineNumber++;
				}
				// build message Certis
				if (!msgCertis.isEmpty()){
					doit(msgCertis, Integer.parseInt(msgCertis.substring(3,5)));
				}
				msgCertis="";
				index++;
				
				if (str==null) eof=true;
			}
			// close XML CERTIS result 
			in.close();
	    	closeXML();
	    	//System.out.println(writer.toString());
	    	ReturnValue[0]= writer.toString();
	    	ReturnValue[1]= RejectMsg;
		} 
		catch (FileNotFoundException e) 
		{
			responseCode = "FAILURE";
			responseDescription = "file : "+ fileName+" doesn't exists";
			loggerBW.error(responseDescription + "\n" + e.getMessage());
			//e.printStackTrace();
		}
		catch (Exception e) 
		{
			responseCode = "FAILURE";
			responseDescription = e.getMessage();
			loggerBW.error(responseDescription);
			//e.printStackTrace();
		} 
		finally
		{
			return ReturnValue;
		}
		
		

	}

	/**
	 * this method build an XML one Message Certis
	 * @param Message
	 * @param msgType
	 */
	private  void doit( String Message , int msgType) {
		String previouselt = null;

		try {
			// get the type of the message Certis
			String nodeName0 = null;
			/**
			 * Changed for Release V3.1 
			 switch (msgType) {
			 	case 01: case 11: case 12: case 13:
			 	case 14:
			 		nodeName0 = "CP";break;
			 	case 15 : case 16: case 17: case 18: case 25: 
			 	case 26:
			 		nodeName0 = "TP";break;
			 	case 21 :
			 		nodeName0 = "B2B";break;
			 	case 33 :
			 		nodeName0 = "RR" ;break;
			 	case 32 :
			 		nodeName0 = "RD";break;
			 	case 55 :
			 		nodeName0 = "RJ";break;
			 	case 61: case 62: case 63: case 64: case 65: case 66: case 67: case 68: 
			 	case 69: case 71: case 72: case 73: case 74: case 75: case 77: case 82: case 83: 
			 	case 85 : 
			 		nodeName0 = "NACK";break;
			 default : nodeName0=null;break;
			 */
			switch (msgType) {
			 	case 01: case 11: case 12: case 13:	case 14: case 96 : case 97 : // CP
			 	case 15 : case 16: case 17: case 18: case 25: case 26: // TP
			 	case 21 : // B2B
			 		nodeName0 = "Payment";
			 		break;
			 	case 33 : //RR
			 	case 32 : //RD
				case 98 : //RD
			 		nodeName0 = "Request" ;
			 		break;
			 	case 55 : //RJ
			 		nodeName0 = "Reject";
			 		break;
			 	case 61: case 62: case 63: case 64: case 65: case 66: case 67: case 68: case 69: 
			 	case 71: case 72: case 73: case 74: case 75: case 77: case 82: case 83:	case 85: case 88 :
			 		nodeName0 = "NACK";
			 		break;
			 	case 52: //Balance --> Added this node to handle Balance Files i.e. HD:52 
			 		nodeName0 = "Balance";
			 		break;
				 default : 
					 nodeName0=null;
				 break;
			}
			
			// build one xml Certis 
			if (nodeName0!=null ){
				 atts.clear();
				 th.startElement("","MESSAGECERTIS","MESSAGECERTIS",atts);
				 th.startElement("",nodeName0,nodeName0,atts);
				 StringTokenizer st = new StringTokenizer(Message, DELIM);
				 if(nodeName0.equals("Reject")){
					// System.out.println("Message is : "+Message);
					 RejectMsg=RejectMsg+"\n"+Message;
					// System.out.println("Complete Msg: "+RejectMsg);
			      }
				 previouselt ="";
				 while (st.hasMoreTokens()) {
					 String str = st.nextToken();
					 
					 if (!str.startsWith("   ")&& !previouselt.isEmpty()) {
						 process( previouselt);
						 previouselt=str;
					 }else
						 previouselt+= previouselt.isEmpty() ?str:DELIM+str;
				
				 } 
				 process(previouselt);
				 th.endElement("",nodeName0,nodeName0);
				 th.endElement("","MESSAGECERTIS","MESSAGECERTIS");
			}
			
		} catch (Exception e) {
			responseCode = "FAILURE";
			responseDescription = e.getMessage();
			//loggerBW.error(e.getMessage());
			//e.printStackTrace();
		}
	}

	/**
	 *   this method initialize  an xml message Certis
	 * @throws ParserConfigurationException
	 * @throws TransformerConfigurationException
	 * @throws SAXException
	 */
	private   void initXML() throws ParserConfigurationException,
			TransformerConfigurationException, SAXException {
		// JAXP + SAX
		System.setProperty("javax.xml.transform.TransformerFactory",
                   "com.sun.org.apache.xalan.internal.xsltc.trax.TransformerFactoryImpl");
       SAXTransformerFactory tf = (SAXTransformerFactory) SAXTransformerFactory.newInstance();
		th = tf.newTransformerHandler();
		Transformer serializer = th.getTransformer();
		serializer.setOutputProperty(OutputKeys.MEDIA_TYPE, "XML");
		//serializer.setOutputProperty(OutputKeys.ENCODING, "ISO-8859-1");
		serializer.setOutputProperty(OutputKeys.ENCODING, getProperty(WRITER_ENCODING));
		// pretty XML output
		 serializer.setOutputProperty
	        ("{http://xml.apache.org/xslt}indent-amount", "4");
	    serializer.setOutputProperty(OutputKeys.INDENT,"yes");

        th.setResult(out);
		th.startDocument();
		
		atts = new AttributesImpl();
		atts.addAttribute("", "xmlns", "xmlns", "CDATA", MESSAGE_XML_NAMESPACE_URI);
		th.startElement("" ,"MESSAGE" , "MESSAGE" , atts);
		atts.addAttribute("", "xmlns", "xmlns", "CDATA", MSGSCERTIS_XML_NAMESPACE_URI);
		th.startElement("" ,"MSGSCERTIS"  ,  "MSGSCERTIS" , atts);
		atts.clear();
		
	}
	
	

	/**
	 * Close an xml Certis result
	 * @throws SAXException
	 */
	private  void closeXML() throws SAXException {
		
	    th.endElement("","MSGSCERTIS","MSGSCERTIS");
	    
	    th.startElement("" ,"POSITION"  ,  "POSITION" , atts);
	    th.characters(String.valueOf(lineNumber).toCharArray(), 0, String.valueOf(lineNumber).length());
	    th.endElement("","POSITION","POSITION");
	    
	    th.startElement("" ,"EOF"  ,  "EOF" , atts);
	    th.characters(String.valueOf(eof).toCharArray(), 0, String.valueOf(eof).length());
	    th.endElement("","EOF","EOF");
	    
	    th.startElement("" ,"ResponseCode"  ,  "ResponseCode" , atts);
	    th.characters(responseCode.toCharArray(), 0, responseCode.length());
	    th.endElement("","ResponseCode","ResponseCode");
	    
	    th.startElement("" ,"ResponseDescription"  ,  "ResponseDescription" , atts);
	    th.characters(responseDescription.toCharArray(), 0, responseDescription.length());
	    th.endElement("","ResponseDescription","ResponseDescription");
	    
	    th.endElement("","MESSAGE","MESSAGE");
	    th.endDocument();
	}

	public  String padStart(String str, int size) {
		return String.format("%" + size + "s", str);
	}
